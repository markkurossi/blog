Code Samples

package main

import (
    "crypto/ed25519"
)

type Garbler struct {
    msg       [64]byte
    privShare [64]byte
}

func main(g Garbler, privShare [64]byte) []byte {
    var priv [64]byte

    for i := 0; i < len(priv); i++ {
        priv[i] = g.privShare[i] ^ privShare[i]
    }

    return ed25519.Sign(priv, g.msg)
}
